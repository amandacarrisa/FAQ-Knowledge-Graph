#Find FAQs with the base verb "aman" which means to protect, along with other FAQs that have its synonyms.

PREFIX ex: <http://example.org/> 
PREFIX ex-verb: <http://example.org/verb/> 
PREFIX ex-noun: <http://example.org/noun/> 
PREFIX ex-adj: <http://example.org/adj/> 
PREFIX ex-class: <http://example.org/class/> 
PREFIX ex-faq: <http://example.org/faq/>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#> 
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> 
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#> 
PREFIX dct: <http://purl.org/dc/terms/>


SELECT DISTINCT ?faqTitle
WHERE {
	{ 
    ?faq dct:subject ex-verb:aman .
    ?faq dct:title ?faqTitle . 
    }
  UNION
    {
     ?synonym skos:exactMatch ex-verb:aman .
     ?faq dct:subject ?synonym .
     ?faq dct:title ?faqTitle . 
    }   
}
